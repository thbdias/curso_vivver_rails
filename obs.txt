.rotas  
  .tipos de rotas
    .resources :"nome do recurso"
        .ex recurso: kind (editar, mostrar, deletar, criar)
    ."verbo" "path (controller/ação)"
        .ex get 'home/index'
    .mapeamento
        .ex get 'home/bemvindo' => 'home#index'
    .only [:ação, :ação]
    .except [:ação]
